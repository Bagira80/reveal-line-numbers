AJAX Framework Overview

First take a step back and review the AJAX Framework.

What is the AJAX Framework?

AJAX framework was introduced in Drupal 7. To Dynamically update data. Allows events to trigger callbacks to the server. A PHP macro language, returns json object with javascript instructions. Json can contain markup or array of commands.

  • Introduced in Drupal 7.
  • Binds events to trigger server callback functions.
  • Server returns a json object of data/commands.

More info: http://bit.ly/DrupalAJAX

Render Array [D7]


/**
 * Custom AJAX callback.
 */
function my_module_ajax_callback($id) {
  // Get news article based on id.
  $article = my_module_get_news_article($id);
  // Return AJAX render array of commands.
  return array(
    '#type' => 'ajax',
    '#commands' => array(
      // Insert news article.
      ajax_command_insert('#news', drupal_render($article)),
      // Add news article to news list.
      ajax_command_append('#news-list', '<li>' . $article->title . '</li>'),
      // Restripe news lists.
      ajax_command_restripe('#news-list'),
      // Invoke JS function to highlight new article.
      ajax_command_invoke(NULL, 'highlightNews', array($id)),
    ),
  );
}

ajax_render() [D7]


/**
 * Custom AJAX callback.
 */
function my_module_ajax_callback($id) {
  // Get news article based on id.
  $article = my_module_get_news_article($id);
  // Build array of AJAX commands.
  $commands = array(
    // Insert news article.
    ajax_command_insert('#news', drupal_render($article)),
    // Add news article to news list.
    ajax_command_append('#news-list', '<li>' . $article->title . '</li>'),
    // Restripe news lists.
    ajax_command_restripe('#news-list'),
    // Invoke JS function to highlight new article.
    ajax_command_invoke(NULL, 'highlightNews', array($id)),
  );
  //Return rendered commands.
  return ajax_render($commands);
}

AJAXResponse() [D8]


/**
 * Custom AJAX callback. (D8)
 */
function my_module_ajax_callback($id) {
  // Get news article based on id.
  $article = my_module_get_news_article($id);
  // Build an AJAX response.
  $response = new AjaxResponse();
  // Insert news article.
  $rendered = \Drupal::service('renderer')->render($article);
  $response->addCommand( new InsertCommand('#news', $rendered);
  // Add news article to news list.
  $li = '<li>' . $article->title . '</li>';
  $response->addCommand( new AppendCommand('#news-list', $li));
  // Restripe news lists.
  $response->addCommand( new RestripeCommand('#news-list'));
  // Invoke JS function to highlight new article.
  $response->addCommand( new InvokeCommand(NULL, 'highlightNews', array($id)));
  // Return ajax response.
  return $response;
}
